import { Component, OnInit, ViewChild } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { SelectionModel } from '@angular/cdk/collections';
import {
  MatPaginator,
  MatSort,
  MatDialog,
  MatDialogRef,
  MatTableDataSource,
  MatTable,
  MAT_DIALOG_DATA
} from '@angular/material';

import { SwapItem, BoardService } from '../../core/services/board.service';
import { AddEditItemDialogComponent } from '../add-edit-item-dialog/add-edit-item-dialog.component';
import { ProposeTradeDialogComponent } from '../propose-trade-dialog/propose-trade-dialog.component';

@Component({
  selector: 'app-main-board',
  templateUrl: './main-board.component.html',
  styleUrls: ['./main-board.component.scss']
})
export class MainBoardComponent implements OnInit {
  public title: string;
  public userPictureSrc: string;
  public displayedColumns: string[];
  public swapItems: SwapItem[];
  public swapItemsDS: MatTableDataSource<SwapItem>;
  public selection: any;
  public currentUserName: string;
  @ViewChild(MatPaginator) paginator: MatPaginator;
  @ViewChild(MatSort) sort: MatSort;
  @ViewChild(MatTable) table: MatTable<any>;

  constructor(
    private route: ActivatedRoute,
    private boardService: BoardService,
    private matDialog: MatDialog
  ) {
    this.displayedColumns = [
      'select',
      'name',
      'brewery',
      'style',
      'teir',
      'numAvailable',
      'packageSize',
      'owner',
      'desiredSwap',
      'minTeirAccepted'
    ];
  }

  ngOnInit() {
    this.currentUserName = this.route.snapshot.data.data.name;
    this.swapItems = this.boardService.getAllItems();
    this.swapItemsDS = new MatTableDataSource<SwapItem>(this.swapItems);
    this.swapItemsDS.paginator = this.paginator;
    this.swapItemsDS.sort = this.sort;

    this.selection = new SelectionModel<SwapItem>(true, []);
  }
  public isAllSelected = () => {
    const numSelected = this.selection.selected.length;
    const numRows = this.swapItemsDS.data.length;
    return numSelected === numRows;
  }

  /** Selects all rows if they are not all selected; otherwise clear selection. */
  public masterToggle = () => {
    this.isAllSelected()
      ? this.selection.clear()
      : this.swapItemsDS.data.forEach(row => this.selection.select(row));
  }

  public applyFilter = (inputVal: string) => {
    this.swapItemsDS.filter = inputVal.trim().toLowerCase();
  }

  public showEditPostingBtn = () => {
    return (
      this.selection.selected.length &&
      this.isASelectionOwnedByUser(this.selection)
    );
  }
  public showDeleteBtn = () => {
    return (
      this.selection.selected.length === 1 &&
      this.selection.selected[0].owner === this.currentUserName
    );
  }

  public showProposeBtn = () => {
    return (
      this.selection.selected.length &&
      this.isAtleastOneUnOwnedSelection(this.selection)
    );
  }

  private isASelectionOwnedByUser = selects => {
    for (const item of selects.selected) {
      if (item.owner === this.currentUserName) {
        return true;
      }
    }
    return false;
  }
  private isAtleastOneUnOwnedSelection = selects => {
    for (const item of selects.selected) {
      if (item.owner !== this.currentUserName) {
        return true;
      }
    }
    return false;
  }

  private updateItems = updatedItems => {
    for (const updatedItem of updatedItems) {
      for (let item of this.swapItemsDS.data) {
        if (item.id === updatedItem.id) {
          item = updatedItem;
        }
      }
    }
  }

  private getNextId = () => {
    this.swapItemsDS.data.sort((item1, item2) => {
      return item1.id - item2.id;
    });
    return this.swapItemsDS.data[this.swapItemsDS.data.length - 1].id++;
  }

  public onDeleteItem = selected => {
    if (selected.length === 1) {
      this.boardService.deleteItem(selected.id);
      for (let i = 0; i < this.swapItemsDS.data.length; i++) {
        if (selected[0].id === this.swapItemsDS.data[i].id) {
          this.swapItemsDS.data.splice(i, 1);
          this.swapItemsDS = new MatTableDataSource<SwapItem>(
            this.swapItemsDS.data
          );
          this.swapItemsDS.paginator = this.paginator;
          this.swapItemsDS.sort = this.sort;

          this.selection = new SelectionModel<SwapItem>(true, []);
        }
      }
    } else {
      // must select only one item to delete at a time
      window.alert(
        'Please select only one item to delete at a time; Also you may only delete your posted items'
      );
    }
  }
  public onAddItem = () => {
    this.matDialog
      .open(AddEditItemDialogComponent, {
        width: '35vw',
        data: {
          title: 'Add Item',
          isEdit: false,
          currentUserName: this.currentUserName,
          // tslint:disable-next-line:arrow-return-shorthand
          nextId: this.getNextId()
        }
      })
      .afterClosed()
      .subscribe(result => {
        if (result) {
          this.boardService.addItem(result);
          // this.swapItemsDS.data.push(result);
          this.swapItemsDS = new MatTableDataSource<SwapItem>(
            this.swapItemsDS.data
          );
          this.swapItemsDS.paginator = this.paginator;
          this.swapItemsDS.sort = this.sort;

          this.selection = new SelectionModel<SwapItem>(true, []);
        }
      });
  }
  public onEditItem = () => {
    this.matDialog
      .open(AddEditItemDialogComponent, {
        width: '250px',
        data: {
          title: 'Edit Item(s)',
          isEdit: true,
          items: this.selection.selected.filter((item, index) => {
            return item.owner === this.currentUserName;
          })
        }
      })
      .afterClosed()
      .subscribe(result => {
        this.updateItems(result);
      });
  }
  public onProposeTrade = () => {
    const dialogRef = this.matDialog.open(ProposeTradeDialogComponent, {
      width: '250px',
      data: {
        title: 'Propose Trade',
        items: this.selection.selected.filter((item, index) => {
          return item.owner !== this.currentUserName;
        }),
        ownedItems: this.swapItems.filter((item, index) => {
          return item.owner === this.currentUserName;
        })
      }
    });

    dialogRef.afterClosed().subscribe(result => {
      const proposals = result;
      console.log(proposals);
      // remove proposals from main board (mark them pending trade)
      // Write new proposals to database
      // ??refresh grid??
    });
  }
}
