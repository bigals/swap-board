  public getAllItems = () => {
    return this.mockData;
  }

  public getItem = (id: number) => {
    // tslint:disable-next-line:prefer-const
    for (let item of this.mockData) {
      if (item.id === id) {
        return item;
      }
    }
  }

  public addItem = (item: SwapItem) => {
    // tslint:disable-next-line:arrow-return-shorthand
    this.mockData.sort((item1: SwapItem, item2: SwapItem) => {
      return item1.id - item2.id;
    });
    item.id = this.mockData[this.mockData.length - 1].id + 1;
    this.mockData.push(item);
  }

  public deleteItem = (id: number) => {
    for (let i = 0; i < this.mockData.length; i++) {
      if (this.mockData[i].id === id) {
        this.mockData.splice(i, 1);
        return;
      }
    }
  }

  public editItem = (item: SwapItem) => {
    for (let dbItem of this.mockData) {
      if (dbItem.id === item.id) {
        dbItem = item;
        return;
      }
    }
  }
}
